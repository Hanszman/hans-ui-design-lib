The **Kanban** component provides a drag-and-drop workflow board built without external drag libraries. It is composed of reusable `KanbanColumn` and `KanbanItem` building blocks, while each item internally uses the standalone `Card` component as its visual template.

## Usage

```tsx
import { useState } from 'react';
import { HansKanban } from 'hans-ui-design-lib';

const columns = [
  { id: 'todo', title: 'To do' },
  { id: 'doing', title: 'Doing' },
  { id: 'done', title: 'Done' },
];

const initialItems = [
  { id: 'task-1', columnId: 'todo', title: 'Refine backlog' },
  { id: 'task-2', columnId: 'doing', title: 'Build drag and drop' },
];

export const Example = () => {
  const [items, setItems] = useState(initialItems);

  return (
    <HansKanban
      columns={columns}
      items={items}
      onItemsChange={setItems}
      emptyColumnText="Drop a task here"
    />
  );
};
```
